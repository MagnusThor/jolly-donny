Interface IOfflineStorageProvider

Interface representing an offline storage provider. This interface defines methods for initializing, saving, updating, deleting, and querying data in an offline storage system.

interface IOfflineStorageProvider {
    addCollection<T extends PersistedEntityBase>(
        label: string,
        collection: any,
    ): void;
    all<T extends PersistedEntityBase>(label: string): Promise<T[]>;
    delete<T extends PersistedEntityBase>(
        label: string,
        item: T,
    ): Promise<void>;
    find<
        T extends PersistedEntityBase,
        K extends string | number | symbol = keyof T,
    >(
        label: string,
        query: (item: T) => boolean,
        pickKeys?: K[],
    ): Promise<Pick<T, K>[]>;
    findById<T extends PersistedEntityBase>(
        label: string,
        uuid: string,
    ): Promise<undefined | T>;
    getCollections(): Promise<Map<string, any>>;
    init(storageName: string): Promise<void>;
    save(): Promise<void>;
    update<T extends PersistedEntityBase>(
        label: string,
        item: T,
    ): Promise<void>;
}
